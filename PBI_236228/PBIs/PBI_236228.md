\## Phase 5: Implement Filtering Logic



\### Step 5.1: Create JavaScript Web Resource

1\. In solution \*\*pbi\_236228\*\*, go to \*\*Web Resources\*\* → \*\*New\*\*

2\. Fill:

&#x20;  - \*\*Name:\*\* completeisequestionnaire\_main

&#x20;  - \*\*Display Name:\*\* ISF Questionnaire Main Form Script

&#x20;  - \*\*Type:\*\* Script (JavaScript)

3\. Click \*\*Text Editor\*\* and paste this code:



```javascript

// ISF Questionnaire Form - Deal Director Filtering Script

// PBI 236228: Filter Deal Director field to show only active deal team members



var ISFFormNamespace = ISFFormNamespace || {};



ISFFormNamespace.DealDirectorFilter = {

&#x20;   

&#x20;   // Main function to apply filter on form load

&#x20;   onFormLoad: function() {

&#x20;       console.log("ISF Form Load: Initializing Deal Director filter");

&#x20;       ISFFormNamespace.DealDirectorFilter.applyDealDirectorFilter();

&#x20;   },

&#x20;   

&#x20;   // Apply filter to Deal Director lookup field

&#x20;   applyDealDirectorFilter: function() {

&#x20;       try {

&#x20;           // Get the Deal field value

&#x20;           var dealLookup = Xrm.Page.getAttribute("new\_deal").getValue();

&#x20;           

&#x20;           if (!dealLookup || dealLookup.length === 0) {

&#x20;               console.warn("No deal selected. Cannot filter Deal Director.");

&#x20;               return;

&#x20;           }

&#x20;           

&#x20;           var dealId = dealLookup\[0].id;

&#x20;           console.log("Deal ID: " + dealId);

&#x20;           

&#x20;           // Build FetchXML to get active deal team members for this deal

&#x20;           var fetchXml = `

&#x20;               <fetch version="1.0" output-format="xml-platform" mapping="logical" distinct="false">

&#x20;                   <entity name="new\_dealteammember">

&#x20;                       <attribute name="new\_dealteammemberid" />

&#x20;                       <attribute name="new\_dtmname" />

&#x20;                       <attribute name="new\_membername" />

&#x20;                       <order attribute="new\_dtmname" descending="false" />

&#x20;                       <filter type="and">

&#x20;                           <condition attribute="new\_deal" operator="eq" value="${dealId}" />

&#x20;                           <condition attribute="new\_status" operator="eq" value="100000000" />

&#x20;                       </filter>

&#x20;                   </entity>

&#x20;               </fetch>

&#x20;           `;

&#x20;           

&#x20;           // Apply filter to Deal Director lookup control

&#x20;           var dealDirectorControl = Xrm.Page.getControl("new\_dealdirector");

&#x20;           

&#x20;           if (dealDirectorControl) {

&#x20;               dealDirectorControl.addPreSearch(function() {

&#x20;                   dealDirectorControl.addCustomFilter(fetchXml, "new\_dealteammember");

&#x20;               });

&#x20;               console.log("Deal Director filter applied successfully");

&#x20;           } else {

&#x20;               console.error("Deal Director control not found on form");

&#x20;           }

&#x20;           

&#x20;       } catch (error) {

&#x20;           console.error("Error in applyDealDirectorFilter: " + error.message);

&#x20;       }

&#x20;   },

&#x20;   

&#x20;   // Refresh filter when Deal field changes

&#x20;   onDealChange: function() {

&#x20;       console.log("Deal field changed: Refreshing Deal Director filter");

&#x20;       ISFFormNamespace.DealDirectorFilter.applyDealDirectorFilter();

&#x20;   }

};



// Form Load Event Handler

function ISFForm\_OnLoad(executionContext) {

&#x20;   var formContext = executionContext.getFormContext();

&#x20;   

&#x20;   // Initialize on form load

&#x20;   ISFFormNamespace.DealDirectorFilter.onFormLoad();

&#x20;   

&#x20;   // Add event listener for Deal field changes

&#x20;   formContext.getAttribute("new\_deal").addOnChange(function() {

&#x20;       ISFFormNamespace.DealDirectorFilter.onDealChange();

&#x20;   });

}

```

\## Common Issues \& Solutions



| Issue | Solution |

|---|---|

| Deal Director dropdown shows all members | Check FetchXML filter condition for Status = Active (value: 100000000) |

| Filter not applying | Ensure web resource is added to form OnLoad event |

| Form loads slowly | Reduce size of FetchXML query; add additional filters |

| "No records" in dropdown | Verify DTM records exist with Status = Active for that Deal |

| JavaScript errors in console | Check syntax; use browser DevTools (F12) to debug |

| Deal change doesn't refresh filter | Ensure onDealChange handler is registered properly |
